    document.addEventListener('DOMContentLoaded', function () {
    // Selecting the display element and all buttons
    const display = document.querySelector('.display');
        const buttons = document.querySelectorAll('.button');
    
      // Initializing a variable to store the current input
      let currentInput = '';
    
      // Adding event listeners to all buttons
      buttons.forEach(button => {
          button.addEventListener('click', () => {
              const buttonText = button.textContent;
    
              // Handling different button clicks
              if (buttonText === 'AC') {
                  currentInput = ''; // Clear the input
              } else if (buttonText === '+/-') {
                  currentInput = eval('-' + currentInput); // Negate the input
              } else if (buttonText === '=') {
                  currentInput = eval(currentInput); // Evaluate the expression
              } else {
                  currentInput += buttonText; // Concatenate input
              }
    
              // Update the display with the current input
              display.textContent = currentInput;
          });
      });
    
      // Adding event listener to handle keyboard input
      document.addEventListener('keydown', (event) => {
        const key = event.key;
    
          // Handling different keyboard inputs
          if (!isNaN(parseInt(key))) {
              currentInput += key; // Append number to input
          } else if (key === '+' || key === '-' || key === '*' || key === '/' || key === '.') {
              currentInput += key; // Append operator to input
          } else if (key === 'Enter') {
              currentInput = eval(currentInput); // Evaluate expression on Enter
          } else if (key === 'Backspace') {
              currentInput = currentInput.slice(0, -1); // Remove last character on Backspace
          } else if (key === 'Escape') {
              currentInput = ''; // Clear input on Escape
          }
    
          // Update the display with the current input
          display.textContent = currentInput;
      });
    });

        // Selecting the container for line numbers and the code content

        const codeContent = document.getElementById("code-content");
        // Splitting the code content by lines and adding line numbers to the container
        const lines = codeContent.innerText.split("\n");
        for (let i = 1; i <= lines.length; i++) {
            const lineNumberElement = document.createElement("pre");
            lineNumberElement.setAttribute("data-line", i);
            lineNumberElement.textContent = i;
            lineNumbersContainer.appendChild(lineNumberElement);
        }
    });
